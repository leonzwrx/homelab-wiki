# start full-text-rss (rootless)
podman run -d \
    --network rss-network \
    --name full-text-rss \
    -p 50000:80 \
    -v full-text-rss-config:/config:Z \
    -v full-text-rss-data:/data:Z \
    -v full-text-rss-cache:/var/www/html/cache \
    docker.io/heussd/fivefilters-full-text-rss:latest
